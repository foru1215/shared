            // æ‰‰ã®å…¥åŠ›ãŒã‚ã‚‹ã‹ã©ã†ã‹ã®ãƒã‚§ãƒƒã‚¯
            let hasDoor = false;
            if (isAutoCalcActive) {
                const w = parseFloat(document.getElementById('autoCalcW').value) || 0;
                const h = parseFloat(document.getElementById('autoCalcH').value) || 0;
                hasDoor = w > 0 && h > 0;
            } else {
                const doorFrontW = parseFloat(document.querySelector('#doorFront [data-type="width"]').value) || 0;
                const doorFrontH = parseFloat(document.querySelector('#doorFront [data-type="height"]').value) || 0;
                hasDoor = doorFrontW > 0 && doorFrontH > 0;
            }
            
            if (isAutoCalcActive) {
                const w = parseFloat(document.getElementById('autoCalcW').value) || 0;
                const h = parseFloat(document.getElementById('autoCalcH').value) || 0;
                const d = parseFloat(document.getElementById('autoCalcD').value) || 0;
                const material = document.getElementById('autoCalcMaterialSelect').value;
                const thickness = document.getElementById('autoCalcThicknessSelect').value;

                if (w > 0 && h > 0 && d > 0 && material && thickness) {
                    const density = materialDensity[material][thickness];
                    const weightFactor = density / 1000000;

                    const plateParts = [
                        // æ‰‰ã®å…¥åŠ›ãŒã‚ã‚‹å ´åˆã¯ç®±æ­£é¢ã¨ç®±è£é¢ã¯è¨ˆç®—ã—ãªã„
                        ...(hasDoor ? [] : [{ label: 'ç®±æ­£é¢', area: w * h, count: 1, key: 'box' }]),
                        ...(hasDoor ? [] : [{ label: 'ç®±è£é¢', area: w * h, count: 1, key: 'box' }]),
                        { label: 'ç®±å´é¢', area: d * h, count: 2, key: 'box' },
                        { label: 'ç®±å¤©äº•', area: w * d, count: 1, key: 'box' },
                        { label: 'ç®±åº•é¢', area: w * d, count: 1, key: 'box' },
                        // æ‰‰ã®å…¥åŠ›ãŒã‚ã‚‹å ´åˆã¯æ‰‰æ­£é¢ã¨æ‰‰è£é¢ã‚’è¨ˆç®—
                        ...(hasDoor ? [{ label: 'æ‰‰æ­£é¢', area: w * h, count: 1, key: 'door' }] : []),
                        ...(hasDoor ? [{ label: 'æ‰‰è£é¢', area: w * h, count: 1, key: 'door' }] : []),
                    ];
                    
                    plateParts.forEach(part => {
                        const weight = part.area * weightFactor * part.count;
                        breakdown[part.key] += weight;
                        const dimensionText = (part.label === 'ç®±å´é¢') ? `${d}x${h}` : (part.label.includes('åº•é¢') || part.label.includes('å¤©äº•')) ? `${w}x${d}` : `${w}x${h}`;
                        calculationExpressions.push(`${part.label} (${materialNames[material]} t${thickness}mm): (${dimensionText}) / 1000000 x ${density} x ${part.count} = ${weight.toFixed(1)}kg`);
                    });

                    const equipmentCheckbox = document.getElementById('equipmentCheckbox');
                    if(equipmentCheckbox.checked) {
                        const equipmentWeight = (w * h) * weightFactor;
                        breakdown.equipment = equipmentWeight;
                        calculationExpressions.push(`å™¨å…·æ¿: (${w}x${h}) / 1000000 x ${density} = ${equipmentWeight.toFixed(1)}kg`);
                    }

                    const otherPartsCheckbox = document.getElementById('otherPartsCheckbox');
                    if(otherPartsCheckbox.checked) {
                        const otherWeight = parseFloat(document.getElementById('otherPartsWeightAuto').value) || 0;
                        breakdown.parts = otherWeight;
                        if (otherWeight > 0) {
                            calculationExpressions.push(`éƒ¨å“è³ªé‡ï¼ˆãã®ä»–ï¼‰ï¼š${otherWeight.toFixed(1)}kg`);
                        }
                    }
                }

                const autoFrameRows = document.querySelectorAll('#autoFrameContainer .frame-row');
                autoFrameRows.forEach(row => {
                    const length = parseFloat(row.querySelector('[data-type="length"]')?.value || 0);
                    const count = parseFloat(row.querySelector('[data-type="count"]')?.value || 1);
                    const unitWeight = parseFloat(row.querySelector('.unit-weight-input')?.value || 0);
                    const selectedOptionText = row.querySelector('.frame-selector option:checked')?.textContent;
                    
                    if (length > 0 && unitWeight > 0) {
                        const weight = length * unitWeight * count;
                        breakdown.frame += weight;
                        calculationExpressions.push(`ãƒ•ãƒ¬ãƒ¼ãƒ æ (${selectedOptionText})ï¼š${length}m x ${unitWeight}kg/m x ${count} = ${weight.toFixed(1)}kg`);
                    }
                });

            } else {
                const parts = [
                    { id: 'boxFront', breakdownKey: 'box' },
                    { id: 'boxBack', breakdownKey: 'box' },
                    { id: 'boxSide', breakdownKey: 'box' },
                    { id: 'boxTop', breakdownKey: 'box' },
                    { id: 'boxBottom', breakdownKey: 'box' },
                    { id: 'doorFront', breakdownKey: 'door' },
                    { id: 'doorBack', breakdownKey: 'door' },
                    { id: 'roofFront', breakdownKey: 'roof' },
                    { id: 'roofSide', breakdownKey: 'roof' },
                    { id: 'roofTop', breakdownKey: 'roof' },
                    { id: 'equipment', breakdownKey: 'equipment' },
                ];
                
                parts.forEach(part => {
                    // æ‰‰ã®å…¥åŠ›ãŒã‚ã‚‹å ´åˆã¯ç­ä½“æ­£é¢ã¨ç­ä½“è£é¢ã®è¨ˆç®—ã‚’ã‚¹ã‚­ãƒƒãƒ—
                    if (hasDoor && (part.id === 'boxFront' || part.id === 'boxBack')) {
                        return;
                    }
                    
                    const row = document.getElementById(part.id);
                    if (!row) return;

                    const w = parseFloat(row.querySelector('[data-type="width"]')?.value || 0);
                    const h = parseFloat(row.querySelector('[data-type="height"]')?.value || 0);
                    const d = parseFloat(row.querySelector('[data-type="depth"]')?.value || 0);
                    const count = parseFloat(row.querySelector('[data-type="count"]')?.value || 1);
                    
                    if (w <= 0 || (h <= 0 && d <= 0)) return;

                    const material = row.querySelector('.material-select').value;
                    const thickness = row.querySelector('.thickness-select').value;
                    const density = materialDensity[material][thickness];
                    
                    let area = 0;
                    let dimensionText = '';
                    if (w && h) {
                        area = (w * h) / 1000000;
                        dimensionText = `${w} x ${h}`;
                    } else if (w && d) {
                        area = (w * d) / 1000000;
                        dimensionText = `${w} x ${d}`;
                    }
                    
                    const weight = area * density * count;
                    breakdown[part.breakdownKey] += weight;
                    const label = row.querySelector('.input-label').textContent;
                    calculationExpressions.push(`${label} (${materialNames[material]} t${thickness}mm): (${dimensionText}) / 1000000 x ${density} x ${count} = ${weight.toFixed(1)}kg`);
                });

                const otherWeight = parseFloat(document.getElementById('otherPartsWeightManual').value) || 0;
                breakdown.parts = otherWeight;
                if (otherWeight > 0) {
                    calculationExpressions.push(`éƒ¨å“è³ªé‡ï¼ˆãã®ä»–ï¼‰ï¼š${otherWeight.toFixed(1)}kg`);
                }
                
                const manualFrameRows = document.querySelectorAll('#manualFrameContainer .frame-row');
                manualFrameRows.forEach(row => {
                    const length = parseFloat(row.querySelector('[data-type="length"]')?.value || 0);
                    const count = parseFloat(row.querySelector('[data-type="count"]')?.value || 1);
                    const unitWeight = parseFloat(row.querySelector('.unit-weight-input')?.value || 0);
                    const selectedOptionText = row.querySelector('.frame-selector option:checked')?.textContent;
                    
                    if (length > 0 && unitWeight > 0) {
                        const weight = length * unitWeight * count;
                        breakdown.frame += weight;
                        calculationExpressions.push(`ãƒ•ãƒ¬ãƒ¼ãƒ æ (${selectedOptionText})ï¼š${length}m x ${unitWeight}kg/m x ${count} = ${weight.toFixed(1)}kg`);
                    }
                });
            }

            totalWeight = breakdown.box + breakdown.door + breakdown.roof + 
                         breakdown.equipment + breakdown.frame + breakdown.parts;

            document.getElementById('totalWeight').textContent = totalWeight.toFixed(1) + ' kg';
            document.getElementById('boxWeight').textContent = breakdown.box.toFixed(1) + ' kg';
            document.getElementById('doorWeight').textContent = breakdown.door.toFixed(1) + ' kg';
            document.getElementById('roofWeight').textContent = breakdown.roof.toFixed(1) + ' kg';
            document.getElementById('equipmentWeight').textContent = breakdown.equipment.toFixed(1) + ' kg';
            document.getElementById('frameWeight').textContent = breakdown.frame.toFixed(1) + ' kg';
            document.getElementById('partsWeight').textContent = breakdown.parts.toFixed(1) + ' kg';

            if (calculationExpressions.length > 0) {
                document.getElementById('calculationDetails').innerHTML = '<strong>è¨ˆç®—è©³ç´°:</strong><br>' + calculationExpressions.join('<br>');
            } else {
                document.getElementById('calculationDetails').innerHTML = '';
            }
        }
